def build_category_keyboard() -> InlineKeyboardMarkup:
    """Main menu: one button per category, 2 per row."""
    keyboard = []
    row = []
    for cid, icon, label, _, _ in CATEGORY_DISPLAY:
        row.append(InlineKeyboardButton(f"{icon} {label}", callback_data=f"cat_{cid}"))
        if len(row) == 2:
            keyboard.append(row)
            row = []
    if row:
        keyboard.append(row)
    return InlineKeyboardMarkup(keyboard)


def build_pairs_keyboard(cat_id: str) -> InlineKeyboardMarkup:
    """Pair grid for a category + Back button."""
    icon, label, syms, cols = _CAT_BY_ID[cat_id]
    keyboard = []
    row = []
    for sym in syms:
        # Strip -OTC suffix for display width, keep full name as callback
        display = sym.replace("-OTC", " OTC")
        row.append(InlineKeyboardButton(display, callback_data=f"pair_{sym}"))
        if len(row) == cols:
            keyboard.append(row)
            row = []
    if row:
        keyboard.append(row)
    keyboard.append([InlineKeyboardButton("⬅️ Categories", callback_data="cat_menu")])
    return InlineKeyboardMarkup(keyboard)

class MessageFormatter:

    @staticmethod
    def signal_message(sig: FinalSignal, elapsed: int = 0) -> str:
        """
        Formats a FinalSignal into a Telegram message.

        ENTRY CLARITY:
          pre_close = True  →  ⚡ ENTER NOW  (candle closing, next open imminent)
          pre_close = False →  ⏱ NEXT CANDLE  (enter at open of next candle)
        """
        direction_emoji = "📈" if sig.direction == "CALL" else "📉"
        direction_color = "🟢" if sig.direction == "CALL" else "🔴"

        # ── Confidence bar ────────────────────────────────────────────────────
        filled = int(sig.confidence / 10)
        bar    = "█" * filled + "░" * (10 - filled)

        if sig.confidence >= 85:
            strength = "🔥 EXTREME"
        elif sig.confidence >= 73:
            strength = "💪 STRONG"
        elif sig.confidence >= 62:
            strength = "✅ MODERATE"
        else:
            strength = "⚠️ WEAK"

        # ── ENTRY BLOCK — the most important visual element ───────────────────
        if sig.pre_close:
            secs_left = max(0, 60 - elapsed)
            entry_block = (
                f"┏━━━━━━━━━━━━━━━━━━━━━━━━━━━┓\n"
                f"┃  ⚡⚡  E N T E R   N O W  ⚡⚡  ┃\n"
                f"┗━━━━━━━━━━━━━━━━━━━━━━━━━━━┛\n"
                f"_Place trade immediately_\n"
                f"_Candle closing in ~{secs_left}s — next open is your entry_"
            )
        else:
            entry_block = (
                f"┏━━━━━━━━━━━━━━━━━━━━━━━━━━━┓\n"
                f"┃  ⏱  N E X T  C A N D L E  ⏱  ┃\n"
                f"┗━━━━━━━━━━━━━━━━━━━━━━━━━━━┛\n"
                f"_Wait for current candle to close_\n"
                f"_Enter at the OPEN of the next candle_"
            )

        # ── Confluence block ──────────────────────────────────────────────────
        confluence_text = "\n".join(f"  {f}" for f in sig.confluence_factors)

        msg = (
            f"━━━━━━━━━━━━━━━━━━━━━━━━━━━\n"
            f"{direction_emoji} *BINARY SIGNAL*  {direction_color}\n"
            f"━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n"
            f"📊 *Asset:*    `{sig.asset}`\n"
            f"🎯 *Action:*   *{sig.direction}*  "
            f"({'UP ↑' if sig.direction == 'CALL' else 'DOWN ↓'})\n"
            f"⏱ *Expiry:*  `{sig.expiry}`\n\n"
            f"📐 *Pattern:* {sig.candle_emoji} _{sig.candle_pattern}_\n"
            f"💬 _{sig.candle_description}_\n\n"
            f"📊 *Confidence:* `{sig.confidence}%`  {strength}\n"
            f"`[{bar}]`\n\n"
            f"🔍 *Confluence:*\n"
            f"{confluence_text}\n\n"
            f"🕐 `{sig.timestamp}`\n"
            f"━━━━━━━━━━━━━━━━━━━━━━━━━━━\n"
            f"{entry_block}\n"

class BinarySignalBot:

    SCAN_INTERVAL     = 60   # seconds between auto-scans
    PRE_CLOSE_TRIGGER = 45   # seconds into candle: pre-close window

    def __init__(self, telegram_token: str, twelve_api_key: str):
        self.token     = telegram_token
        self.feed      = TwelveDataFeed(twelve_api_key)
        self.engine    = ConfluenceEngine()
        self.formatter = MessageFormatter()
        self.state     = BotState()
        self.app       = Application.builder().token(telegram_token).job_queue(JobQueue()).build()
        self._register_handlers()

    # ── HANDLER REGISTRATION ──────────────────────────────────────────────────

    def _register_handlers(self):
        self.app.add_handler(CommandHandler("start",  self.cmd_start))
        self.app.add_handler(CommandHandler("pairs",  self.cmd_pairs))
        self.app.add_handler(CommandHandler("help",   self.cmd_help))
        self.app.add_handler(CommandHandler("stats",  self.cmd_stats))
        self.app.add_handler(CommandHandler("stop",   self.cmd_stop))
        self.app.add_handler(CommandHandler("resume", self.cmd_resume))
        self.app.add_handler(CallbackQueryHandler(self.handle_callback))

    # ── STATE HELPERS ─────────────────────────────────────────────────────────

    def _ensure_state(self, chat_id: int):
        if chat_id not in self.state.stats:
            self.state.stats[chat_id] = {"signals_sent": 0, "calls": 0, "puts": 0}

    def _start_scan_job(self, ctx: ContextTypes.DEFAULT_TYPE, chat_id: int):
        """
        Cancel any existing scan job for this chat, then start a new one.
        Called whenever the user selects a new pair.
        """
        job_name = f"scan_{chat_id}"
        # Remove existing jobs for this chat
        for job in ctx.job_queue.get_jobs_by_name(job_name):
            job.schedule_removal()
        # Start fresh scan
        ctx.job_queue.run_repeating(
            self._auto_scan_job,
            interval=self.SCAN_INTERVAL,
            first=self.SCAN_INTERVAL,  # First auto-scan after 60s; immediate is manual
            name=job_name,
            data={"chat_id": chat_id}
        )

    def _record_stat(self, chat_id: int, sig: FinalSignal):
        self._ensure_state(chat_id)
        s = self.state.stats[chat_id]
        s["signals_sent"] += 1
        s["calls"] += (1 if sig.direction == "CALL" else 0)
        s["puts"]  += (1 if sig.direction == "PUT"  else 0)

    # ── CALLBACK HANDLER ─────────────────────────────────────────────────────

    async def handle_callback(self, update: Update, ctx: ContextTypes.DEFAULT_TYPE):
        query   = update.callback_query
        await query.answer()
        data    = query.data
        chat_id = update.effective_chat.id
        self._ensure_state(chat_id)

        try:
            await self._dispatch_callback(query, data, chat_id, ctx)
        except Exception as e:
            logger.error(f"Callback error [{data}] chat={chat_id}: {e}")

    async def _dispatch_callback(self, query, data: str, chat_id: int, ctx):
        """
        All tap actions route through here.

        cat_menu          → show category selector
        cat_{id}          → show pairs for that category
        pair_{symbol}     → select pair, run immediate analysis
        change_pair       → show category selector (from post-signal toggle)
        same_pair         → re-analyse the current active pair immediately
        resume_pick       → unpause + show category selector
        """

        # ── CATEGORY MENU ────────────────────────────────────────────────────
        if data == "cat_menu":
            await query.edit_message_text(
                self.formatter.category_menu_text(),
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── CATEGORY SELECTED → show pairs ───────────────────────────────────
        elif data.startswith("cat_"):
            cat_id = data[4:]
            if cat_id not in _CAT_BY_ID:
                return
            icon, label, syms, cols = _CAT_BY_ID[cat_id]
            await query.edit_message_text(
                self.formatter.pairs_menu_text(icon, label, len(syms)),
                reply_markup=build_pairs_keyboard(cat_id),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── PAIR SELECTED → set active, run immediate analysis ───────────────
        elif data.startswith("pair_"):
            symbol = data[5:]
            await self._select_pair_and_analyse(query, chat_id, symbol, ctx)

        # ── CHANGE PAIR (post-signal toggle) → category selector ─────────────
        elif data == "change_pair":
            await query.edit_message_text(
                self.formatter.category_menu_text(),
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── SAME PAIR (post-signal toggle) → re-analyse immediately ──────────
        elif data == "same_pair":
            symbol = self.state.active_pair.get(chat_id)
            if not symbol:
                # Somehow no active pair — send them back to selector
                await query.edit_message_text(
                    "⚠️ No active pair found.\n_Pick a pair to continue:_",
                    reply_markup=build_category_keyboard(),
                    parse_mode=ParseMode.MARKDOWN
                )
                return
            # Show scanning indicator on the toggle message, then send new signal
            await query.edit_message_text(
                f"⏳ *Re-scanning `{symbol}`...*",
                parse_mode=ParseMode.MARKDOWN
            )
            await self._run_analysis_and_send(chat_id, symbol, ctx.bot)

        # ── RESUME + PICK PAIR ────────────────────────────────────────────────
        elif data == "resume_pick":
            self.state.paused.discard(chat_id)
            await query.edit_message_text(
                "▶️ *Resumed!*\n\n_Pick a pair to start scanning:_",
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

    # ── PAIR SELECTION FLOW ───────────────────────────────────────────────────

    async def _select_pair_and_analyse(
        self,
        query,
        chat_id: int,
        symbol:  str,
        ctx:     ContextTypes.DEFAULT_TYPE
    ):
        """
        Called when user taps a pair button.
        1. Set as active pair
        2. Show "Analysing..." in the menu message
        3. Run analysis
        4. Send signal (or no-signal) as new message with post-signal toggle
        5. Start / restart the 60s auto-scan job
        """
        self.state.active_pair[chat_id] = symbol
        self.state.paused.discard(chat_id)

        # Show scanning indicator in the menu message
        await query.edit_message_text(
            f"⏳ *Analysing `{symbol}`...*\n"
            f"_Scanning M1 candles — please wait_",
            parse_mode=ParseMode.MARKDOWN

    async def _run_analysis_and_send(self, chat_id: int, symbol: str, bot):
        """
        Core: analyse one symbol and send the result to the chat.

        """
        signal  = await self._analyse_asset(symbol)
        elapsed = await self.feed.get_current_seconds_into_minute()

        if signal:
            self._record_stat(chat_id, signal)
            try:
                await bot.send_message(
                    chat_id    = chat_id,
                    text       = self.formatter.signal_message(signal, elapsed),
                    parse_mode = ParseMode.MARKDOWN,
                    reply_markup = build_post_signal_keyboard()
                )
            except Exception as e:
                logger.error(f"Send signal error to {chat_id}: {e}")
        else:
            try:
                await bot.send_message(
                    chat_id    = chat_id,
                    text       = self.formatter.no_signal_message(symbol),
                    parse_mode = ParseMode.MARKDOWN,
                    reply_markup = build_post_signal_keyboard()
                )
            except Exception as e:
                logger.error(f"Send no-signal error to {chat_id}: {e}")

    # ── AUTO-SCAN JOB ─────────────────────────────────────────────────────────

    async def _auto_scan_job(self, ctx: ContextTypes.DEFAULT_TYPE):
        """
        Background repeating job — fires every SCAN_INTERVAL seconds.
        Scans the user's single active pair only.
        """
        chat_id = ctx.job.data["chat_id"]

        if chat_id in self.state.paused:
            return

        symbol = self.state.active_pair.get(chat_id)
        if not symbol:
            return

        signal  = await self._analyse_asset(symbol)
        elapsed = await self.feed.get_current_seconds_into_minute()

        if signal:
            self._record_stat(chat_id, signal)
            try:
                await ctx.bot.send_message(
                    chat_id      = chat_id,
                    text         = self.formatter.signal_message(signal, elapsed),
                    parse_mode   = ParseMode.MARKDOWN,
                    reply_markup = build_post_signal_keyboard()
                )
            except Exception as e:
                logger.error(f"Auto-scan send error to {chat_id}: {e}")
        # No signal on auto-scan = silent (don't flood with no-signal messages)

    # ── CORE ANALYSIS PIPELINE ────────────────────────────────────────────────

    async def _analyse_asset(self, symbol: str) -> Optional[FinalSignal]:
        """Full pipeline: resolve symbol → fetch candles → analyse → return signal."""
        try:
            fetch_symbol = resolve_symbol(symbol)

            raw_candles = await self.feed.fetch_candles(
                fetch_symbol, interval="1min", count=60
            )
            if len(raw_candles) < 10:
                return None

            candles = [
                Candle(
                    open   = c["open"],
                    high   = c["high"],
                    low    = c["low"],
                    close  = c["close"],
                    volume = c.get("volume", 0.0)
                )
                for c in raw_candles
            ]

            raw_forming = await self.feed.fetch_live_quote(fetch_symbol)
            forming = Candle(
                open   = raw_forming["open"],
                high   = raw_forming["high"],
                low    = raw_forming["low"],
                close  = raw_forming["close"],
                volume = raw_forming.get("volume", 0.0)
            ) if raw_forming else None

            elapsed = await self.feed.get_current_seconds_into_minute()
            ts      = datetime.now(timezone.utc).strftime("%Y-%m-%d %H:%M:%S UTC")

            return self.engine.evaluate(
                asset           = symbol,
                candles         = candles,
                forming_candle  = forming,
                elapsed_seconds = elapsed,
                timestamp       = ts
            )

        except Exception as e:
            logger.error(f"Analysis error [{symbol}]: {e}")
            return None

# ─────────────────────────────────────────────────────────────────────────────
# CANDLE PARSER — handles Twelve Data's string-typed fields
# ─────────────────────────────────────────────────────────────────────────────

def _safe_float(value, fallback: float = 0.0) -> float:
    """
    Safely cast a Twelve Data field value to float.
    API returns all OHLCV values as strings — this handles:
      "148.372"  → 148.372
      None       → fallback
      ""         → fallback
      "N/A"      → fallback
    """
    if value is None:
        return fallback
    try:
        result = float(str(value).strip())
        return result if result == result else fallback  # NaN guard
    except (ValueError, TypeError):
        return fallback


def _parse_candle(raw: dict, symbol: str = "") -> dict:
    """
    Parse a single Twelve Data candle from the 'values' array.

    Raw format (all strings):
    {
        "datetime": "2021-08-26 15:59:00",
        "open":     "148.372",
        "high":     "148.405",
        "low":      "148.365",
        "close":    "148.390",
        "volume":   "2206"        ← string, may be "0" or absent for forex
    }

    Returns a properly typed dict with a guaranteed 'volume' float key.
    """
    volume_raw = raw.get("volume")

    # Volume handling:
    # - Forex pairs: "0" is common (tick volume not always provided)
    # - Missing key: treat as 0 and let the engine handle gracefully
    volume = _safe_float(volume_raw, fallback=VOLUME_FALLBACK)

    return {
        "datetime": raw.get("datetime", ""),
        "open":     _safe_float(raw.get("open")),
        "high":     _safe_float(raw.get("high")),
        "low":      _safe_float(raw.get("low")),
        "close":    _safe_float(raw.get("close")),
        "volume":   volume,
    }


def _parse_candles(values: list, symbol: str = "") -> list:
    """
    Parse and order a full candle list from Twelve Data.

    Twelve Data returns values in DESCENDING order (newest first).
    We reverse to ascending (oldest first) for correct EMA/analysis.
    """
    parsed = [_parse_candle(c, symbol) for c in values]
    parsed.reverse()   # ascending: oldest → newest
    return parsed


def _assess_volume_quality(candles: list) -> dict:
    """
    Analyse the quality of volume data in a candle set.

    Returns a dict with:
      - has_volume:    bool — enough non-zero volume for spike analysis
      - zero_pct:      float — percentage of candles with zero volume
      - avg_volume:    float — mean of non-zero volumes
      - recommendation: str — human-readable quality note
    """
    volumes = [c["volume"] for c in candles]
    non_zero = [v for v in volumes if v > 0]
    zero_pct = (len(volumes) - len(non_zero)) / max(len(volumes), 1) * 100

    has_volume = len(non_zero) >= MIN_VOLUME_CANDLES
    avg_volume = sum(non_zero) / len(non_zero) if non_zero else 0.0

    if zero_pct >= 80:
        rec = "Volume unavailable for this asset — spike analysis disabled"
    elif zero_pct >= 40:
        rec = "Sparse volume data — spike signals may be unreliable"
    else:
        rec = "Volume data adequate for spike analysis"

    return {
        "has_volume":      has_volume,
        "zero_pct":        zero_pct,
        "avg_volume":      avg_volume,
        "recommendation":  rec,
    }

class TwelveDataFeed:
    """
    Async HTTP client for Twelve Data API.

    Provides:
      fetch_candles()      — closed M1 candles with volume, ascending order
      fetch_live_quote()   — forming (live) candle from /quote endpoint
      get_current_seconds_into_minute() — timing for pre-close detection
      validate_api_key()   — startup health check
    """

    BASE_URL = "https://api.twelvedata.com"

    def __init__(self, api_key: str):
        self.api_key = api_key
        self._session: Optional[aiohttp.ClientSession] = None

    async def _get_session(self) -> aiohttp.ClientSession:
        if self._session is None or self._session.closed:
            timeout = aiohttp.ClientTimeout(total=10)
            self._session = aiohttp.ClientSession(timeout=timeout)
        return self._session

    async def close(self):
        if self._session and not self._session.closed:
            await self._session.close()

    # ── CANDLE FETCH ──────────────────────────────────────────────────────────

    async def fetch_candles(
        self,
        symbol:   str,
        interval: str = "1min",
        count:    int = 60,
    ) -> list:
        """
        Fetch `count` closed M1 candles for `symbol`.

        Returns a list of candle dicts ordered oldest→newest:
        [
            {"datetime": ..., "open": float, "high": float,
             "low": float, "close": float, "volume": float},
            ...
        ]

        Volume notes:
        - Stocks/Crypto/Commodities: real traded volume (float)
        - Forex: tick volume (number of price ticks per minute)
        - Both are valid for relative spike detection in CRT analysis
        - Volume "0" values are preserved (engine handles gracefully)
        """
        session = await self._get_session()

        params = {
            "symbol":     symbol,
            "interval":   interval,
            "outputsize": count,
            "apikey":     self.api_key,
            "order":      "desc",   # newest first (we reverse after parsing)
            "format":     "JSON",
        }

        try:
            async with session.get(f"{self.BASE_URL}/time_series", params=params) as resp:
                if resp.status != 200:
                    logger.error(f"[{symbol}] HTTP {resp.status} from Twelve Data")
                    return []

                data = await resp.json()

                # API-level error check
                if data.get("status") == "error":
                    msg = data.get("message", "Unknown error")
                    logger.error(f"[{symbol}] Twelve Data error: {msg}")
                    return []

                values = data.get("values", [])
                if not values:
                    logger.warning(f"[{symbol}] No candle data returned")
                    return []

                # Parse with type coercion + ascending sort
                candles = _parse_candles(values, symbol)

                # Log volume quality for diagnostics
                vq = _assess_volume_quality(candles)
                if vq["zero_pct"] > 40:
                    logger.warning(
                        f"[{symbol}] Volume quality: {vq['zero_pct']:.0f}% zero — "
                        f"{vq['recommendation']}"
                    )
                else:
                    logger.debug(
                        f"[{symbol}] Volume quality OK — avg={vq['avg_volume']:.0f}, "
                        f"zero={vq['zero_pct']:.0f}%"
                    )

                return candles

        except asyncio.TimeoutError:
            logger.error(f"[{symbol}] Timeout fetching candles")
            return []
        except Exception as e:
            logger.error(f"[{symbol}] fetch_candles error: {e}")
            return []

    # ── LIVE QUOTE (FORMING CANDLE) ───────────────────────────────────────────

    async def fetch_live_quote(self, symbol: str) -> Optional[dict]:
        """
        Fetch the current live/forming candle from the /quote endpoint.

        Twelve Data /quote returns the CURRENT open candle, giving us
        a real-time snapshot for pre-close analysis.

        Returns a candle dict with volume (may be partial — that's expected).
        """
        session = await self._get_session()

        params = {
            "symbol": symbol,
            "apikey": self.api_key,
        }

        try:
            async with session.get(f"{self.BASE_URL}/quote", params=params) as resp:
                if resp.status != 200:
                    logger.error(f"[{symbol}] HTTP {resp.status} from /quote")
                    return None

                data = await resp.json()

                if data.get("status") == "error":
                    logger.error(f"[{symbol}] Quote error: {data.get('message')}")
                    return None

                # /quote response uses same field names — all strings
                # Fields: open, high, low, close, volume, previous_close, etc.
                return {
                    "datetime": data.get("datetime", ""),
                    "open":     _safe_float(data.get("open")),
                    "high":     _safe_float(data.get("high")),
                    "low":      _safe_float(data.get("low")),
                    "close":    _safe_float(data.get("close")),
                    "volume":   _safe_float(data.get("volume"), fallback=VOLUME_FALLBACK),
                }

        except asyncio.TimeoutError:
            logger.error(f"[{symbol}] Timeout on /quote")
            return None
        except Exception as e:
            logger.error(f"[{symbol}] fetch_live_quote error: {e}")
            return None

    # ── TIMING ────────────────────────────────────────────────────────────────

    async def get_current_seconds_into_minute(self) -> int:
        """
        Returns how many seconds into the current minute we are (0–59).
        Used by the CRT engine's pre-close detection (40–55s window).

        Based on UTC time to match Twelve Data's candle timestamps.
        """
        return datetime.now(timezone.utc).second

    # ── API KEY VALIDATION ────────────────────────────────────────────────────

    async def validate_api_key(self) -> str:
        """
        Validate the API key by hitting the /api_usage endpoint.

        Returns one of three strings:
          "ok"           — key is valid and has credits available
          "rate_limited" — key is valid but hit per-minute credit cap
                           (common on free tier if bot ran recently)
          "invalid"      — key is wrong, expired, or not recognised
        """
        session = await self._get_session()
        try:
            async with session.get(
                f"{self.BASE_URL}/api_usage",
                params={"apikey": self.api_key}
            ) as resp:
                data = await resp.json()

                # HTTP 200 + status ok = all good
                if resp.status == 200 and data.get("status") == "ok":
                    return "ok"

                # Rate limit message — key is valid, just throttled
                message = data.get("message", "") or data.get("code", "")
                if "run out of API credits" in str(message) or "429" in str(resp.status):
                    logger.warning(f"API key validation: rate limited — {message}")
                    return "rate_limited"

                # Anything else (401, wrong key, etc.) = invalid
                logger.error(f"API key validation failed: {message}")
                return "invalid"

        except Exception as e:
            logger.error(f"API key validation error: {e}")
            # Network error — assume rate limited not invalid (safer default)
            return "rate_limited"

async def check_volume_availability(
    feed:    TwelveDataFeed,
    symbols: list[str],
) -> dict[str, dict]:
    """
    Run a volume quality check across a list of symbols at startup.

    Usage in bot.py:
        report = await check_volume_availability(self.feed, ["EUR/USD", "BTC/USD", ...])
        for symbol, info in report.items():

                "has_volume":      False,
                "zero_pct":        100.0,
                "avg_volume":      0.0,
                "recommendation":  "Could not fetch candles",
            }
    return results

        if any(k in base for k in ["SPX", "NDX", "DJI", "FTSE", "DAX", "NAS", "SP500"]):
            return "index"

        # Stock identifiers — uppercase single-word tickers without "/"

    def evaluate(
        self,

        timestamp: str = ""

        if len(candles) < 5:

            forming_candle  = forming_dict,
            elapsed_seconds = int(elapsed_seconds),
        )

        # ── Layer 3: Direction conflict check ─────────────────────────────────
        if candle_sig and crt_sig:
            if candle_sig.direction != crt_sig.direction:
                return None   # Layers conflict — skip this candle

        if candle_sig is None and crt_sig is None:
            return None

        lead_direction = (candle_sig.direction if candle_sig
                          else crt_sig.direction)

        confluence = []
        boost      = 0.0

        if lead_direction == "CALL" and levels.trend == "BULLISH":
            confluence.append("✅ EMA trend aligns BULLISH (9 > 21 > 50)")
            boost += 0.08
        elif lead_direction == "PUT" and levels.trend == "BEARISH":
            confluence.append("✅ EMA trend aligns BEARISH (9 < 21 < 50)")
            boost += 0.08
        elif levels.trend == "SIDEWAYS":
            confluence.append("⚠️ Sideways trend — lower reliability")
            boost -= 0.04
        else:
            confluence.append("❌ Counter-trend signal — use caution")
            boost -= 0.10

        if levels.ema_9 != 0 and levels.ema_21 != 0:
            gap = abs(levels.ema_9 - levels.ema_21) / current.close * 100
            if gap > 0.05:
                confluence.append(f"✅ EMA spread {gap:.3f}% — trending momentum")
                boost += 0.04
            else:
                confluence.append(f"⚠️ EMAs flat ({gap:.3f}%) — choppy market")
                boost -= 0.02

        at_level, level_desc = self.at_key_level(current.close, levels)
        if at_level:
            confluence.append(f"✅ {level_desc} — highest-probability zone")
            boost += 0.12

        if candle_sig and candle_sig.sub_patterns:
            extras = ", ".join(candle_sig.sub_patterns)
            confluence.append(f"✅ Confirmed by: {extras}")
            boost += 0.05 * len(candle_sig.sub_patterns)

        if current.body_ratio > 0.55:
            confluence.append(f"✅ Strong candle body ({current.body_ratio:.0%}) — high conviction")
            boost += 0.04
        elif current.body_ratio < 0.20:
            confluence.append("⚠️ Small candle body — weak conviction")
            boost -= 0.03

            crt_boost = (crt_sig.confidence / 100) * 0.15
            boost += crt_boost

        base_conf = candle_sig.confidence if candle_sig else (crt_sig.confidence / 100)
        raw       = base_conf + boost

        # ── Layer 4: ACCURACY FILTER SUITE  ── NEW BLOCK ─────────────────────
        # ═════════════════════════════════════════════════════════════════════
        #
        # Run all five filters AFTER the core confluence layers.
        # The filters adjust the raw confidence up or down and can suppress
        # the signal entirely (e.g. dead market, extreme RSI, dead session).

        # closes is extracted from the Candle objects list.
        # is_otc is detected from the asset name suffix.
        #
        is_otc     = asset.endswith("-OTC")
        asset_type = self._get_asset_type(asset)
        closes     = [c.close for c in candles]

            asset        = asset,
            asset_type   = asset_type,
            direction    = lead_direction,
            candle_dicts = candle_dicts,
            closes       = closes,
            cooldown     = self.cooldown,
            is_otc       = is_otc,
        )

        # Hard suppress — dead market, extreme RSI, news spike, dead session
        if filter_result.suppress:
            return None

        # Apply filter delta to raw confidence
        raw += filter_result.confidence_delta

        # Append filter factors to confluence block (shows in Telegram message)
        if filter_result.factors:
            confluence.append("━━━ ACCURACY FILTERS ━━━")
            confluence.extend(filter_result.factors)
        # ═════════════════════════════════════════════════════════════════════

        # ── Final confidence gate ─────────────────────────────────────────────
        final_pct = int(max(0, min(raw, 0.97)) * 100)

        # ── Smart Expiry Calculation ────────────────────────────
        ema_gap = abs(levels.ema_9 - levels.ema_21) / current.close * 100

            confidence = final_pct,
            trend      = levels.trend,
            ema_gap    = ema_gap,
            body_ratio = current.body_ratio,
            crt_sig    = crt_sig,
            pre_close  = pre_close,
            is_otc     = is_otc
        )

        if final_pct < self.SIGNAL_THRESHOLD:
            return None

        # ── Pattern label ─────────────────────────────────────────────────────
        if candle_sig and crt_sig:
            pattern     = f"{candle_sig.pattern} + CRT {crt_sig.sweep_side.replace('_', ' ').title()}"
            description = candle_sig.description
            emoji       = candle_sig.emoji + " 📐"
        elif crt_sig:
            pattern     = f"CRT {crt_sig.sweep_side.replace('_', ' ').title()}"
            description = (f"CRT sweep detected — Vol spike {crt_sig.volume.volume_ratio:.1f}x | "
                           f"Range: {crt_sig.zone.range_low:.5f}–{crt_sig.zone.range_high:.5f}")
            emoji       = "📐⚡"
        else:
            pattern     = candle_sig.pattern
            description = candle_sig.description
            emoji       = candle_sig.emoji

        # ── Entry note ────────────────────────────────────────────────────────
        if pre_close:
            entry_note = f"⚡ PRE-CLOSE ALERT at {elapsed_seconds:.0f}s — Enter NOW on next candle open"
        else:
            entry_note = "Enter at open of NEXT candle after current closes"

        # ── Build final signal ────────────────────────────────────────────────
        signal = FinalSignal(
            asset=asset,
            direction=lead_direction,
            confidence=final_pct,
            candle_pattern=pattern,
            candle_description=description,
            candle_emoji=emoji,
            confluence_factors=confluence,
            entry_note=entry_note,
            expiry=expiry,
            timestamp=timestamp,
            pre_close=pre_close
        )

        # ── NEW: Record cooldown AFTER signal is confirmed ────────────────────
        # Called here so the cooldown tracker knows this asset/direction fired.
        # Prevents chasing the same direction on the next candle scan.
        self.cooldown.record(asset, lead_direction)
        # ─────────────────────────────────────────────────────────────────────

        return signal

    @staticmethod
    def calculate_stochastic(
        candles:  list,
        k_period: int = 10,
        k_smooth: int = 3,
        d_period: int = 5,
    ) -> tuple:
        """
        Full Slow Stochastic (K_period=10, K_smooth=3, D_period=5).

        Fast %K  = (Close − LowestLow_N) / (HighestHigh_N − LowestLow_N) × 100
        Slow %K  = SMA(Fast %K, k_smooth)   ← the line traders call %K
        %D       = SMA(Slow %K, d_period)   ← signal / confirmation line

        Accepts a list of dicts with keys: open, high, low, close.

        Returns (slow_k, d) — both 0-100.
        Returns (50.0, 50.0) if not enough data (neutral — no effect on signal).
        """
        needed = k_period + k_smooth + d_period - 2
        if len(candles) < needed:
            return 50.0, 50.0

        # ── Build fast %K series ──────────────────────────────────────────
        fast_k_series = []
        for i in range(k_period - 1, len(candles)):
            window     = candles[i - k_period + 1 : i + 1]
            lowest_low = min(c["low"]  for c in window)
            highest_hi = max(c["high"] for c in window)
            denom      = highest_hi - lowest_low
            if denom == 0:
                fast_k_series.append(50.0)
            else:
                fast_k_series.append(
                    (candles[i]["close"] - lowest_low) / denom * 100
                )

        # ── Slow %K = rolling SMA(k_smooth) of fast %K ───────────────────
        slow_k_series = [
            statistics.mean(fast_k_series[j - k_smooth + 1 : j + 1])
            for j in range(k_smooth - 1, len(fast_k_series))
        ]

        # ── %D = rolling SMA(d_period) of slow %K ────────────────────────
        d_series = [
            statistics.mean(slow_k_series[j - d_period + 1 : j + 1])
            for j in range(d_period - 1, len(slow_k_series))
        ]

        if not slow_k_series or not d_series:
            return 50.0, 50.0

        return round(slow_k_series[-1], 2), round(d_series[-1], 2)

            confluence.append("✅ EMA trend aligns BEARISH (9 < 21 < 50)")
            boost += 0.08
        elif levels.trend == "SIDEWAYS":
            confluence.append("⚠️ Sideways trend — lower reliability")
            boost -= 0.04
        else:
            confluence.append("❌ Counter-trend signal — use caution")
            boost -= 0.10

        if levels.ema_9 != 0 and levels.ema_21 != 0:
            gap = abs(levels.ema_9 - levels.ema_21) / current.close * 100
            if gap > 0.05:
                confluence.append(f"✅ EMA spread {gap:.3f}% — trending momentum")
                boost += 0.04
            else:
                confluence.append(f"⚠️ EMAs flat ({gap:.3f}%) — choppy market")
                boost -= 0.02

        at_level, level_desc = self.at_key_level(current.close, levels)
        if at_level:
            confluence.append(f"✅ {level_desc} — highest-probability zone")
            boost += 0.12

        if candle_sig and candle_sig.sub_patterns:
            extras = ", ".join(candle_sig.sub_patterns)
            confluence.append(f"✅ Confirmed by: {extras}")
            boost += 0.05 * len(candle_sig.sub_patterns)

        if current.body_ratio > 0.55:
            confluence.append(f"✅ Strong candle body ({current.body_ratio:.0%}) — high conviction")
            boost += 0.04
        elif current.body_ratio < 0.20:
            confluence.append("⚠️ Small candle body — weak conviction")
            boost -= 0.03

        if crt_sig:
            confluence.append("━━━ CRT + VOLUME LAYER ━━━")
            confluence.extend(crt_sig.confluence_factors)
            crt_boost = (crt_sig.confidence / 100) * 0.15
            boost += crt_boost

        # ── Base confidence (pre-filter) ──────────────────────────────────
        base_conf = candle_sig.confidence if candle_sig else (crt_sig.confidence / 100)
        raw       = base_conf + boost

        # ═════════════════════════════════════════════════════════════════
        # ── Layer 5: STOCHASTIC (10, 3, 5)  ← NEW v2.1 ───────────────────
        # ═════════════════════════════════════════════════════════════════
        stoch_k, stoch_d = self.calculate_stochastic(candle_dicts)

        confluence.append("━━━ STOCHASTIC (10,3,5) ━━━")

        if lead_direction == "CALL":
            if stoch_k < 20 and stoch_d < 20:
                confluence.append(
                    f"✅ Oversold — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → CALL confirmed"
                )
                boost += 0.07
            elif stoch_k < 35:
                confluence.append(
                    f"☑️ Low zone — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → mild CALL support"
                )
                boost += 0.03
            elif stoch_k > 75:
                confluence.append(
                    f"⚠️ Overbought — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → CALL risk"
                )
                boost -= 0.06
            else:
                confluence.append(
                    f"➡️ Neutral — %K={stoch_k:.1f}  %D={stoch_d:.1f}"
                )

        else:  # PUT
            if stoch_k > 80 and stoch_d > 80:
                confluence.append(
                    f"✅ Overbought — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → PUT confirmed"
                )
                boost += 0.07
            elif stoch_k > 65:
                confluence.append(
                    f"☑️ High zone — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → mild PUT support"
                )
                boost += 0.03
            elif stoch_k < 25:
                confluence.append(
                    f"⚠️ Oversold — %K={stoch_k:.1f}  %D={stoch_d:.1f}  → PUT risk"
                )
                boost -= 0.06
            else:
                confluence.append(
                    f"➡️ Neutral — %K={stoch_k:.1f}  %D={stoch_d:.1f}"
                )

        # %K / %D crossover — only fires in mid-zone to avoid double-counting extremes
        if lead_direction == "CALL" and stoch_k > stoch_d and 20 <= stoch_k <= 60:
            confluence.append(
                f"✅ %K crossed above %D — bullish momentum building"
            )
            boost += 0.03
        elif lead_direction == "PUT" and stoch_k < stoch_d and 40 <= stoch_k <= 80:
            confluence.append(
                f"✅ %K crossed below %D — bearish momentum building"
            )
            boost += 0.03
        # ═════════════════════════════════════════════════════════════════

        # ── Session detection  ← NEW v2.1 ────────────────────────────────
        session_label, session_quality = self._get_current_session()

        # ═════════════════════════════════════════════════════════════════
        # ── Layer 4: ACCURACY FILTER SUITE ───────────────────────────────
        # ═════════════════════════════════════════════════════════════════
        is_otc     = asset.endswith("-OTC")
        asset_type = self._get_asset_type(asset)
        closes     = [c.close for c in candles]

        filter_result = run_all_filters(
            asset        = asset,
            asset_type   = asset_type,
            direction    = lead_direction,
            candle_dicts = candle_dicts,
            closes       = closes,
            cooldown     = self.cooldown,
            is_otc       = is_otc,
        )

        # Hard suppress — dead market, extreme RSI, news spike, dead session
        if filter_result.suppress:
            return None

        # Apply filter delta to raw confidence
        raw += filter_result.confidence_delta

        # Append filter factors to confluence block
        if filter_result.factors:
            confluence.append("━━━ ACCURACY FILTERS ━━━")
            confluence.extend(filter_result.factors)

        # ── Final confidence gate ─────────────────────────────────────────
        final_pct = int(max(0, min(raw, 0.97)) * 100)

        if final_pct < self.SIGNAL_THRESHOLD:
            return None

        # ── Pattern label ─────────────────────────────────────────────────
        if candle_sig and crt_sig:
            pattern     = f"{candle_sig.pattern} + CRT {crt_sig.sweep_side.replace('_', ' ').title()}"
            description = candle_sig.description
            emoji       = candle_sig.emoji + " 📐"
        elif crt_sig:
            pattern     = f"CRT {crt_sig.sweep_side.replace('_', ' ').title()}"
            description = (f"CRT sweep detected — Vol spike {crt_sig.volume.volume_ratio:.1f}x | "
                           f"Range: {crt_sig.zone.range_low:.5f}–{crt_sig.zone.range_high:.5f}")
            emoji       = "📐⚡"
        else:
            pattern     = candle_sig.pattern
            description = candle_sig.description
            emoji       = candle_sig.emoji

        # ── Entry note ────────────────────────────────────────────────────
        if pre_close:
            entry_note = f"⚡ PRE-CLOSE ALERT at {elapsed_seconds:.0f}s — Enter NOW on next candle open"
        else:
            entry_note = "Enter at open of NEXT candle after current closes"

        # ── Build final signal ────────────────────────────────────────────
        signal = FinalSignal(
            asset              = asset,
            direction          = lead_direction,
            confidence         = final_pct,
            candle_pattern     = pattern,
            candle_description = description,
            candle_emoji       = emoji,
            confluence_factors = confluence,
            entry_note         = entry_note,
            expiry             = "M1",
            timestamp          = timestamp,
            pre_close          = pre_close,
            session            = session_label,   # ← NEW v2.1
        )

        # Record cooldown AFTER signal is confirmed — prevents chasing
        self.cooldown.record(asset, lead_direction)

        return signal

        direction_emoji = "📈" if sig.direction == "CALL" else "📉"
        direction_color = "🟢" if sig.direction == "CALL" else "🔴"

        # ── Confidence bar ────────────────────────────────────────────────
        filled = int(sig.confidence / 10)
        bar    = "█" * filled + "░" * (10 - filled)

        if sig.confidence >= 85:
            strength = "🔥 EXTREME"
        elif sig.confidence >= 73:
            strength = "💪 STRONG"
        elif sig.confidence >= 62:
            strength = "✅ MODERATE"
        else:
            strength = "⚠️ WEAK"

        # ── Session line (NEW v2.1) ────────────────────────────────────────
        # sig.session is always populated by signal_engine._get_current_session()
        # Falls back gracefully to empty string if somehow missing.
        session_line = f"🕐 *Session:*  {sig.session}\n" if sig.session else ""

        # ── ENTRY BLOCK ───────────────────────────────────────────────────
        if sig.pre_close:
            secs_left = max(0, 60 - elapsed)
            entry_block = (
                f"┏━━━━━━━━━━━━━━━━━━━━━━━━━━━┓\n"

                fvg_lines.append(line)

        confluence_text = (
            "\n".join(f"  {f}" for f in fvg_lines)
            if fvg_lines else "  _No FVG structure_"
        )

        msg = (
            f"━━━━━━━━━━━━━━━━━━━━━━━━━━━\n"
            f"{direction_emoji} *BINARY SIGNAL*  {direction_color}\n"
            f"━━━━━━━━━━━━━━━━━━━━━━━━━━━\n\n"
            f"📊 *Asset:*    `{sig.asset}`\n"
            f"{session_line}"
            f"🎯 *Action:*   *{sig.direction}*  "
            f"({'UP ↑' if sig.direction == 'CALL' else 'DOWN ↓'})\n"
            f"⏱ *Expiry:*  `{sig.expiry}`\n\n"
            f"📐 *Pattern:* {sig.candle_emoji} _{sig.candle_pattern}_\n"

class BinarySignalBot:

    SCAN_INTERVAL     = 60   # seconds between auto-scans
    PRE_CLOSE_TRIGGER = 45   # seconds into candle: pre-close window

    def __init__(self, telegram_token: str, twelve_api_key: str):
        self.token  = telegram_token
        po_ssid     = os.getenv("PO_SSID", "")
        po_demo     = os.getenv("PO_DEMO", "true").lower() == "true"

        # Store config — don't connect yet
        if po_ssid:
            self.feed = PocketOptionFeed(ssid=po_ssid, demo=po_demo)
        else:
            self.feed = TwelveDataFeed(twelve_api_key)

        self.engine    = ConfluenceEngine()
        self.formatter = MessageFormatter()
        self.state     = BotState()
        self.app       = Application.builder().token(telegram_token).job_queue(JobQueue()).build()
        self._register_handlers()
 
    async def run(self):
        logger.info("🤖 Binary Signal Robot v2.1 starting...")

        # ✅ Connect feed HERE where await is valid
        if isinstance(self.feed, PocketOptionFeed):
            await self.feed.connect()

        try:
            await self.app.initialize()
            await self.app.start()
            await self.app.updater.start_polling(drop_pending_updates=True)
            logger.info("🚀 Bot is live — waiting for commands...")
            await asyncio.Event().wait()
        finally:
            try:
                await self.app.updater.stop()
                await self.app.stop()
                await self.app.shutdown()
            except Exception:
                pass
            await self.feed.close()

    # ── HANDLER REGISTRATION ──────────────────────────────────────────────

    def _register_handlers(self):
        self.app.add_handler(CommandHandler("start",  self.cmd_start))
        self.app.add_handler(CommandHandler("pairs",  self.cmd_pairs))
        self.app.add_handler(CommandHandler("help",   self.cmd_help))
        self.app.add_handler(CommandHandler("stats",  self.cmd_stats))
        self.app.add_handler(CommandHandler("stop",   self.cmd_stop))
        self.app.add_handler(CommandHandler("resume", self.cmd_resume))
        self.app.add_handler(CallbackQueryHandler(self.handle_callback))

    # ── STATE HELPERS ─────────────────────────────────────────────────────

    def _ensure_state(self, chat_id: int):
        if chat_id not in self.state.stats:
            self.state.stats[chat_id] = {"signals_sent": 0, "calls": 0, "puts": 0}

    def _start_scan_job(self, ctx: ContextTypes.DEFAULT_TYPE, chat_id: int):
        """
        Cancel any existing scan job for this chat, then start a new one.

    async def _dispatch_callback(self, query, data: str, chat_id: int, ctx):
        """
        All tap actions route through here.

        cat_menu          → show category selector
        cat_{id}          → show pairs for that category
        pair_{symbol}     → select pair, run immediate analysis
        change_pair       → show category selector (from post-signal toggle)
        same_pair         → re-analyse the current active pair immediately
        resume_pick       → unpause + show category selector
        """

        # ── CATEGORY MENU ────────────────────────────────────────────────
        if data == "cat_menu":
            await query.edit_message_text(
                self.formatter.category_menu_text(),
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── CATEGORY SELECTED → show pairs ───────────────────────────────
        elif data.startswith("cat_"):
            cat_id = data[4:]
            if cat_id not in _CAT_BY_ID:
                return
            icon, label, syms, cols = _CAT_BY_ID[cat_id]
            await query.edit_message_text(
                self.formatter.pairs_menu_text(icon, label, len(syms)),
                reply_markup=build_pairs_keyboard(cat_id),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── PAIR SELECTED → set active, run immediate analysis ───────────
        elif data.startswith("pair_"):
            symbol = data[5:]
            await self._select_pair_and_analyse(query, chat_id, symbol, ctx)

        # ── CHANGE PAIR (post-signal toggle) → category selector ─────────
        elif data == "change_pair":
            await query.edit_message_text(
                self.formatter.category_menu_text(),
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

        # ── SAME PAIR (post-signal toggle) → re-analyse immediately ──────
        elif data == "same_pair":
            symbol = self.state.active_pair.get(chat_id)
            if not symbol:
                await query.edit_message_text(
                    "⚠️ No active pair found.\n_Pick a pair to continue:_",
                    reply_markup=build_category_keyboard(),
                    parse_mode=ParseMode.MARKDOWN
                )
                return
            await query.edit_message_text(
                f"⏳ *Re-scanning `{symbol}`...*",
                parse_mode=ParseMode.MARKDOWN
            )
            await self._run_analysis_and_send(chat_id, symbol, ctx.bot)

        # ── RESUME + PICK PAIR ────────────────────────────────────────────
        elif data == "resume_pick":
            self.state.paused.discard(chat_id)
            await query.edit_message_text(
                "▶️ *Resumed!*\n\n_Pick a pair to start scanning:_",
                reply_markup=build_category_keyboard(),
                parse_mode=ParseMode.MARKDOWN
            )

        # No signal on auto-scan = silent (don't flood with no-signal messages)

    async def _analyse_asset(self, symbol: str) -> Optional[FinalSignal]:
        """Full pipeline: resolve symbol → fetch candles → analyse → return signal."""
        try:
            fetch_symbol = symbol if isinstance(self.feed, PocketOptionFeed) else resolve_symbol(symbol)

            raw_candles = await self.feed.fetch_candles(
                fetch_symbol, interval="1min", count=60
            )
            if len(raw_candles) < 10:
                return None

            candles = [
                Candle(
                    open   = c["open"],
                    high   = c["high"],
                    low    = c["low"],
                    close  = c["close"],
                    volume = c.get("volume", 0.0)
                )
                for c in raw_candles
            ]

            raw_forming = await self.feed.fetch_live_quote(fetch_symbol)
            forming = Candle(
                open   = raw_forming["open"],
                high   = raw_forming["high"],
                low    = raw_forming["low"],
                close  = raw_forming["close"],
                volume = raw_forming.get("volume", 0.0)
            ) if raw_forming else None

            elapsed = await self.feed.get_current_seconds_into_minute()
            ts      = datetime.now(timezone.utc).strftime("%Y-%m-%d %H:%M:%S UTC")

            return self.engine.evaluate(
                asset           = symbol,
                candles         = candles,
                forming_candle  = forming,
                elapsed_seconds = elapsed,
                timestamp       = ts
            )

        except Exception as e:
            logger.error(f"Analysis error [{symbol}]: {e}")
            return None

def _decode(msg) -> str:
    if isinstance(msg, (bytes, bytearray)):
        return msg.decode("utf-8", errors="replace")
    return msg


def _normalize(symbol: str) -> str:
    return (
        symbol.lstrip("#")
        .replace("_otc", "")
        .replace("-OTC", "")
        .replace("-otc", "")
        .replace("/", "")
        .replace(" ", "")
        .upper()
    )


class PocketOptionFeed:

        if ssid:

    # ─────────────────────────────────────────────

    async def fetch_candles(
        self,
        symbol: str,
        interval: str = "1min",
        count: int = 60,
    ) -> list:

        asset_id = self.OTC_ASSETS.get(symbol)

        if asset_id is None:
            logger.error(f"Unknown asset: {symbol}")
            return []

        period = {
            "1min": 60,
            "1m": 60,
            "5min": 300,
            "15min": 900,
        }.get(interval, 60)

        for attempt in range(2):

            try:

                async with await self._open_authenticated_ws() as ws:

                    logger.info(
                        f"📡 Requesting candles for "
                        f"{symbol} "
                        f"(asset={asset_id}, period={period})"
                    )

                    candles = await self._request_candles(
                        ws,
                        asset_id,
                        period,
                        count
                    )

                    if candles:
                        return candles

            except Exception as e:

                logger.error(
                    f"fetch_candles failed: {e}",
                    exc_info=True
                )

            await asyncio.sleep(3)

        return []

    # ─────────────────────────────────────────────
    # WEBSOCKET
    # ─────────────────────────────────────────────

    async def _open_authenticated_ws(self):

        headers = {
            "User-Agent": (
                "Mozilla/5.0 "
                "(Windows NT 10.0; Win64; x64)"
            ),
            "Origin": "https://pocketoption.com",
            "Referer": "https://pocketoption.com/",
        }

        ws = await websockets.connect(
            self.ws_url,
            additional_headers=headers,
            ping_interval=None,
            open_timeout=15,
            close_timeout=5,
            max_size=None,
        )

        msg = _decode(
            await asyncio.wait_for(
                ws.recv(),
                timeout=10
            )
        )

        logger.info(f"[OPEN] {repr(msg)}")

        if not msg.startswith("0"):
            raise Exception(f"Bad EIO open packet: {msg}")

        await ws.send("40")

        namespace_ok = False

        for _ in range(10):

            msg = _decode(
                await asyncio.wait_for(
                    ws.recv(),
                    timeout=10
                )
            )

            logger.info(
                f"[NAMESPACE] {repr(msg)}"
            )

            if msg == "2":
                await ws.send("3")
                continue

            if msg.startswith("40"):
                namespace_ok = True
                break

        if not namespace_ok:
            raise Exception("Namespace connect failed")

        await ws.send(self._raw_ssid)

        logger.info(f"Auth sent (uid={self.uid})")

        self.keepalive_task = asyncio.create_task(
            self._keepalive(ws)
        )

        await self._capture_asset_map(ws)

        await asyncio.sleep(2)

        return ws

    # ─────────────────────────────────────────────

    async def _keepalive(self, ws):

        try:

            while True:

                await asyncio.sleep(20)

                try:
                    await ws.send("2")
                    logger.debug("Ping sent")

                except:
                    break

        except asyncio.CancelledError:
            pass

    # ─────────────────────────────────────────────

    async def _capture_asset_map(self, ws):

        expecting_assets = False

        for _ in range(50):

            try:

                msg = _decode(
                    await asyncio.wait_for(
                        ws.recv(),
                        timeout=5
                    )
                )

            except asyncio.TimeoutError:
                continue

            except ConnectionClosed:
                logger.warning(
                    "WS closed during auth"
                )
                return

            logger.info(
                f"[AUTH PHASE] {repr(msg[:200])}"
            )

            if msg == "2":
                await ws.send("3")
                continue

            if (
                msg.startswith("451-")
                and "updateAssets" in msg
            ):
                expecting_assets = True
                continue

            if expecting_assets and msg.startswith("[["):

                self._build_asset_map(msg)

                logger.info(
                    "✅ Asset map captured"
                )

                return

    # ─────────────────────────────────────────────

    def _build_asset_map(self, payload_str: str):

        try:
            assets = json.loads(payload_str)

        except Exception as e:
            logger.error(
                f"Asset parse failed: {e}"
            )
            return

        new_map = {}

        for entry in assets:

            if not isinstance(entry, list):
                continue

            if len(entry) < 2:
                continue

            asset_id = entry[0]
            po_symbol = str(entry[1])

            norm = _normalize(po_symbol)

            if norm in self._NORM_TO_SYMBOL:

                canonical = (
                    self._NORM_TO_SYMBOL[norm]
                )

                new_map[canonical] = asset_id

        if new_map:

            self.OTC_ASSETS.update(new_map)

            logger.info(
                f"✅ Asset map updated "
                f"({len(new_map)} assets)"
            )

    # ─────────────────────────────────────────────
    # CANDLES
    # ─────────────────────────────────────────────

    async def _request_candles(
        self,
        ws,
        asset_id: int,
        period: int,
        count: int,
    ) -> list:

        change_symbol = (
            '42["changeSymbol",{'
            f'"asset":{asset_id},'
            f'"period":{period}'
            '}]'
        )

        history_request = (
            '42["loadHistoryPeriod",{'
            f'"asset":{asset_id},'
            f'"period":{period},'
            f'"time":{int(time.time())},'
            '"index":0,'
            '"offset":0'
            '}]'
        )

        logger.info(f"→ {change_symbol}")
        await ws.send(change_symbol)

        await asyncio.sleep(2)

        logger.info(f"→ {history_request}")
        await ws.send(history_request)

        deadline = time.time() + 30

        try:

            while time.time() < deadline:

                try:

                    raw = await asyncio.wait_for(
                        ws.recv(),
                        timeout=5
                    )

                    msg = _decode(raw)

                    logger.info(
                        f"[FULL RAW] {repr(msg)}"
                    )

                except asyncio.TimeoutError:
                    continue

                except ConnectionClosed:

                    logger.warning(
                        "WS closed by PocketOption"
                    )

                    break

                if msg == "2":
                    await ws.send("3")
                    continue

                # ─────────────────────
                # 451 events
                # ─────────────────────

                if msg.startswith("451-"):

                    try:

                        json_part = (
                            msg[msg.index("["):]
                        )

                        payload = json.loads(
                            json_part
                        )

                        event = payload[0]

                        logger.info(
                            f"[451 EVENT] {event}"
                        )

                    except:
                        continue

                    continue

                # ─────────────────────
                # Binary attachment
                # ─────────────────────

                if msg.startswith("[["):

                    candles = self._parse_candles(
                        msg,
                        count
                    )

                    if candles:

                        logger.info(
                            f"✅ Parsed "
                            f"{len(candles)} candles "
                            f"from binary attachment"
                        )

                        if self.keepalive_task:
                            self.keepalive_task.cancel()

                        return candles

                # ─────────────────────
                # 42 events
                # ─────────────────────

                if msg.startswith("42"):

                    try:

                        payload = json.loads(
                            msg[2:]
                        )

                    except:
                        continue

                    if not isinstance(payload, list):
                        continue

                    if len(payload) < 2:
                        continue

                    event = payload[0]
                    data = payload[1]

                    logger.info(
                        f"← EVENT: {event}"
                    )

                    if event in self._HISTORY_EVENTS:

                        candles = self._parse_candles(
                            data,
                            count
                        )

                        if candles:

                            logger.info(
                                f"✅ Parsed "
                                f"{len(candles)} candles"
                            )

                            if self.keepalive_task:
                                self.keepalive_task.cancel()

                            return candles

        finally:

            if self.keepalive_task:
                self.keepalive_task.cancel()

        logger.warning(
            "❌ Candle request timed out"
        )

        return []

    # ─────────────────────────────────────────────
    # PARSER
    # ─────────────────────────────────────────────

    def _parse_candles(
        self,
        data,
        count: int
    ) -> list:

        candles = []

        try:

            if isinstance(data, str):

                try:
                    data = json.loads(data)
                except:
                    pass

            raw = None

            if isinstance(data, list):
                raw = data

            elif isinstance(data, dict):

                raw = (
                    data.get("candles")
                    or data.get("history")
                    or data.get("data")
                    or data.get("result")
                )

            if not raw:
                return []

            for c in raw[-count:]:

                # ─────────────────────
                # Dict candle
                # ─────────────────────

                if isinstance(c, dict):

                    try:

                        candles.append({
                            "open": float(
                                c.get("open", c.get("o", 0))
                            ),
                            "high": float(
                                c.get("high", c.get("h", 0))
                            ),
                            "low": float(
                                c.get("low", c.get("l", 0))
                            ),
                            "close": float(
                                c.get("close", c.get("c", 0))
                            ),
                            "volume": float(
                                c.get("volume", c.get("v", 0))
                            ),
                        })

                    except:
                        continue

                # ─────────────────────
                # List candle
                # ─────────────────────

                elif (
                    isinstance(c, (list, tuple))
                    and len(c) >= 5
                ):

                    # FIX v8.3
                    # Validate candle structure

                    try:

                        open_price = float(c[1])
                        high_price = float(c[2])
                        low_price = float(c[3])
                        close_price = float(c[4])

                    except (ValueError, TypeError):

                        logger.warning(
                            f"Skipping non-candle payload: "
                            f"{repr(c[:10])}"
                        )

                        continue

                    candles.append({
                        "open": open_price,
                        "high": high_price,
                        "low": low_price,
                        "close": close_price,
                        "volume": (
                            float(c[5])
                            if (
                                len(c) > 5
                                and isinstance(
                                    c[5],
                                    (int, float)
                                )
                            )
                            else 0.0
                        ),
                    })

        except Exception as e:

            logger.error(
                f"Candle parse failed: {e}",
                exc_info=True
            )

        return candles

@dataclass
class TechnicalLevels:
    ema_9:              float = 0.0
    ema_21:             float = 0.0
    ema_50:             float = 0.0
    support_levels:     list  = field(default_factory=list)
    resistance_levels:  list  = field(default_factory=list)

@dataclass
class FinalSignal:
    asset:               str
    direction:           str           # CALL | PUT
    confidence:          int           # 0–100

# ─────────────────────────────────────────────────────────────────────────────
# CONFLUENCE ENGINE
# ─────────────────────────────────────────────────────────────────────────────

class ConfluenceEngine:
    """
    Four-layer signal engine for M1 binary options.

    All layers must agree on direction for highest confidence.
    Any hard conflict between layers suppresses the signal.
    """

    SIGNAL_THRESHOLD = 62   # minimum confidence % to emit a signal

    def __init__(self):
        self.psychology = CandlePsychologyEngine()
        self.crt        = CRTVolumeEngine(min_confidence=55)
        self.cooldown   = SignalCooldownTracker()

    # ─────────────────────────────────────────────────────────────────────────
    # ASSET TYPE HELPER
    # ─────────────────────────────────────────────────────────────────────────

    def _get_asset_type(self, asset: str) -> str:
        base = asset.replace("-OTC", "").upper()
        if any(k in base for k in ["BTC", "ETH", "BNB", "SOL", "XRP"]):
            return "crypto"
        if any(k in base for k in ["XAU", "XAG", "GOLD", "SILVER", "OIL", "USOIL", "UKOIL"]):
            return "commodity"
        if any(k in base for k in ["SPX", "NDX", "DJI", "FTSE", "DAX", "NAS", "SP500"]):
            return "index"
        if "/" not in base and len(base) <= 5 and base.isalpha():
            return "stock"
        return "forex"

    # ─────────────────────────────────────────────────────────────────────────
    # EMA + LEVELS
    # ─────────────────────────────────────────────────────────────────────────

    def calculate_ema(self, prices: list, period: int) -> float:
        if len(prices) < period:
            return prices[-1] if prices else 0.0
        k   = 2 / (period + 1)
        ema = statistics.mean(prices[:period])
        for price in prices[period:]:
            ema = price * k + ema * (1 - k)
        return ema

    def get_levels(self, candles: list) -> TechnicalLevels:
        if not candles:
            return TechnicalLevels()

        closes = [c.close for c in candles]
        highs  = [c.high  for c in candles]
        lows   = [c.low   for c in candles]

        ema9  = self.calculate_ema(closes, 9)
        ema21 = self.calculate_ema(closes, 21)
        ema50 = self.calculate_ema(closes, min(50, len(closes)))

        resistance = sorted(set(round(h, 5) for h in sorted(highs)[-10:]))
        support    = sorted(set(round(l, 5) for l in sorted(lows)[:10]))

        if ema9 > ema21 > ema50:
            trend = "BULLISH"
        elif ema9 < ema21 < ema50:
            trend = "BEARISH"
        else:
            trend = "SIDEWAYS"

        return TechnicalLevels(
            ema_9=ema9, ema_21=ema21, ema_50=ema50,
            support_levels=support, resistance_levels=resistance, trend=trend
        )

    def at_key_level(self, price: float, levels: TechnicalLevels) -> tuple:
        tolerance = price * 0.0008
        for r in levels.resistance_levels:
            if abs(price - r) <= tolerance:
                return True, f"At Resistance ({r:.5f})"
        for s in levels.support_levels:
            if abs(price - s) <= tolerance:
                return True, f"At Support ({s:.5f})"
        for ema_val, ema_name in [
            (levels.ema_9,  "EMA 9"),
            (levels.ema_21, "EMA 21"),
            (levels.ema_50, "EMA 50"),
        ]:
            if abs(price - ema_val) <= tolerance * 2:
                return True, f"At {ema_name} ({ema_val:.5f})"
        return False, ""

    def _candles_to_dicts(self, candles: list) -> list:
        return [
            {"open": c.open, "high": c.high,
             "low":  c.low,  "close": c.close, "volume": c.volume}
            for c in candles
        ]

    # ─────────────────────────────────────────────────────────────────────────
    # MAIN EVALUATE
    # ─────────────────────────────────────────────────────────────────────────

    def evaluate(
        self,
        asset:           str,
        candles:         list,
        forming_candle:  Optional[Candle] = None,
        elapsed_seconds: float            = 0.0,
        timestamp:       str              = ""
    ) -> Optional[FinalSignal]:

        if len(candles) < 5:
            return None

        levels  = self.get_levels(candles)
        current = candles[-1]

        # ═════════════════════════════════════════════════════════════════════
        # LAYER 1 — Candle Psychology
        # ═════════════════════════════════════════════════════════════════════
        candle_sig = self.psychology.analyse(candles[-5:])
        pre_close  = False

        if forming_candle and elapsed_seconds >= 40:
            live_sig = self.psychology.predict_forming_candle(
                forming_candle, elapsed_seconds
            )
            if live_sig and (candle_sig is None or
                             live_sig.confidence > candle_sig.confidence * 0.85):
                candle_sig = live_sig
                pre_close  = True

        # ═════════════════════════════════════════════════════════════════════
        # LAYER 2 — CRT + Volume
        # ═════════════════════════════════════════════════════════════════════
        candle_dicts = self._candles_to_dicts(candles)
        forming_dict = (
            {
                "open":   forming_candle.open,
                "high":   forming_candle.high,
                "low":    forming_candle.low,
                "close":  forming_candle.close,
                "volume": forming_candle.volume,
            }
            if forming_candle else None
        )

        crt_sig = self.crt.analyse(
            asset           = asset,
            candles         = candle_dicts,
            forming_candle  = forming_dict,
            elapsed_seconds = int(elapsed_seconds),
        )

        # ═════════════════════════════════════════════════════════════════════
        # LAYER 3 — Direction conflict check (candle ↔ CRT)
        # ═════════════════════════════════════════════════════════════════════
        if candle_sig and crt_sig:
            if candle_sig.direction != crt_sig.direction:
                return None

        if candle_sig is None and crt_sig is None:
            return None

        lead_direction = (candle_sig.direction if candle_sig
                          else crt_sig.direction)

        # ═════════════════════════════════════════════════════════════════════
        # EMA + S/R confluence scoring
        # ═════════════════════════════════════════════════════════════════════
        confluence = []
        boost      = 0.0

        if lead_direction == "CALL" and levels.trend == "BULLISH":
            confluence.append("✅ EMA trend aligns BULLISH (9 > 21 > 50)")
            boost += 0.08
        elif lead_direction == "PUT" and levels.trend == "BEARISH":
            confluence.append("✅ EMA trend aligns BEARISH (9 < 21 < 50)")
            boost += 0.08
        elif levels.trend == "SIDEWAYS":
            confluence.append("⚠️ Sideways trend — lower reliability")
            boost -= 0.04
        else:
            confluence.append("❌ Counter-trend signal — use caution")
            boost -= 0.10

        if levels.ema_9 != 0 and levels.ema_21 != 0:
            gap = abs(levels.ema_9 - levels.ema_21) / current.close * 100
            if gap > 0.05:
                confluence.append(f"✅ EMA spread {gap:.3f}% — trending momentum")
                boost += 0.04
            else:
                confluence.append(f"⚠️ EMAs flat ({gap:.3f}%) — choppy market")
                boost -= 0.02

        at_level, level_desc = self.at_key_level(current.close, levels)
        if at_level:
            confluence.append(f"✅ {level_desc} — highest-probability zone")
            boost += 0.12

        if candle_sig and candle_sig.sub_patterns:
            extras = ", ".join(candle_sig.sub_patterns)
            confluence.append(f"✅ Confirmed by: {extras}")
            boost += 0.05 * len(candle_sig.sub_patterns)

        if current.body_ratio > 0.55:
            confluence.append(
                f"✅ Strong candle body ({current.body_ratio:.0%}) — high conviction"
            )
            boost += 0.04
        elif current.body_ratio < 0.20:
            confluence.append("⚠️ Small candle body — weak conviction")
            boost -= 0.03

        if crt_sig:
            confluence.append("━━━ CRT + VOLUME LAYER ━━━")
            confluence.extend(crt_sig.confluence_factors)
            crt_boost = (crt_sig.confidence / 100) * 0.15
            boost += crt_boost

        # ── Base confidence pre-filter ────────────────────────────────────────
        base_conf = (candle_sig.confidence if candle_sig
                     else (crt_sig.confidence / 100))
        raw = base_conf + boost

        # ═════════════════════════════════════════════════════════════════════
        # LAYER 4 — Accuracy Filters (RSI / ATR / Session / Cooldown)
        # ═════════════════════════════════════════════════════════════════════
        is_otc     = asset.endswith("-OTC")
        asset_type = self._get_asset_type(asset)
        closes     = [c.close for c in candles]

        filter_result = run_all_filters(
            asset        = asset,
            asset_type   = asset_type,
            direction    = lead_direction,
            candle_dicts = candle_dicts,
            closes       = closes,
            cooldown     = self.cooldown,
            is_otc       = is_otc,
        )

        if filter_result.suppress:
            return None

        raw += filter_result.confidence_delta

        if filter_result.factors:
            confluence.append("━━━ ACCURACY FILTERS ━━━")
            confluence.extend(filter_result.factors)

        # ── Final confidence gate ─────────────────────────────────────────────
        final_pct = int(max(0, min(raw, 0.97)) * 100)

        # ── Asset-aware confidence band filter ───────────────────────────────
        #
        #   OTC pairs   -> emit ONLY genuine 60-69%
        #                  No clamping — if the engine scores above 69%, reject.
        #                  Only raw setups that naturally land in this band pass.
        #
        #   Forex pairs -> emit 50-69%  OR  > 95%
        #   Everything else -> emit >= SIGNAL_THRESHOLD (62%)
        #
        if is_otc:
            if not (60 <= final_pct <= 69):
                return None
        elif asset_type == "forex":
            if not (50 <= final_pct <= 69 or final_pct > 95):
                return None
        else:
            if final_pct < self.SIGNAL_THRESHOLD:
                return None

        # ── Smart Expiry ──────────────────────────────────────────────────────
        ema_gap = abs(levels.ema_9 - levels.ema_21) / current.close * 100

        expiry = self.determine_expiry(
            confidence = final_pct,
            trend      = levels.trend,
            ema_gap    = ema_gap,
            body_ratio = current.body_ratio,
            crt_sig    = crt_sig,
            pre_close  = pre_close,
            is_otc     = is_otc,
        )

        # ── Pattern label ─────────────────────────────────────────────────────
        if candle_sig and crt_sig:
            pattern     = (
                f"{candle_sig.pattern} + CRT "
                f"{crt_sig.sweep_side.replace('_', ' ').title()}"
            )
            description = candle_sig.description
            emoji       = candle_sig.emoji + " 📐"
        elif crt_sig:
            pattern     = f"CRT {crt_sig.sweep_side.replace('_', ' ').title()}"
            description = (
                f"CRT sweep detected — Vol spike {crt_sig.volume.volume_ratio:.1f}x | "
                f"Range: {crt_sig.zone.range_low:.5f}–{crt_sig.zone.range_high:.5f}"
            )
            emoji       = "📐⚡"
        else:
            pattern     = candle_sig.pattern
            description = candle_sig.description
            emoji       = candle_sig.emoji

        # ── Entry note ────────────────────────────────────────────────────────
        if pre_close:
            entry_note = (
                f"⚡ PRE-CLOSE ALERT at {elapsed_seconds:.0f}s — "
                "Enter NOW on next candle open"
            )
        else:
            entry_note = "Enter at open of NEXT candle after current closes"

        # ── Resolve session label (used by MessageFormatter / bot.py) ──────────
        session_label = getattr(filter_result, "session", "") or ""

        # ── Build final signal ────────────────────────────────────────────────
        signal = FinalSignal(
            asset              = asset,
            direction          = lead_direction,
            confidence         = final_pct,
            candle_pattern     = pattern,
            candle_description = description,
            candle_emoji       = emoji,
            confluence_factors = confluence,
            entry_note         = entry_note,
            expiry             = expiry,
            timestamp          = timestamp,
            pre_close          = pre_close,
            session            = session_label,
        )

        # Record cooldown
        self.cooldown.record(asset, lead_direction)

        return signal

    # ─────────────────────────────────────────────────────────────────────────
    # SMART EXPIRY
    # ─────────────────────────────────────────────────────────────────────────

    def determine_expiry(
        self,
        confidence: int,
        trend:      str,
        ema_gap:    float,
        body_ratio: float,
        crt_sig,
        pre_close:  bool,
        is_otc:     bool,
    ) -> str:
        """
        Smart expiry calculation based on confidence, trend, and CRT signals.
        """
        # Pre-close always 1 min (entering at next candle open)
        if pre_close:
            return "1 MINUTE"

        # High confidence strong trend
        if confidence >= 80 and trend in ("BULLISH", "BEARISH") and ema_gap > 0.05:
            return "2 MINUTES"

        # CRT sweep with high confidence
        if crt_sig and confidence >= 75:
            return "2 MINUTES"

        # Sideways or low confidence = shorter
        if trend == "SIDEWAYS" or confidence < 68:
            return "1 MINUTE"

        # OTC default: 1 min (tighter market, faster reversals)
        if is_otc:
            return "1 MINUTE"

        return "1 MINUTE"

def _safe_float(value, fallback: float = 0.0) -> float:
    """
    Safely cast a Twelve Data field value to float.
    API returns all OHLCV values as strings — this handles:
      "148.372"  → 148.372
      None       → fallback
      ""         → fallback
      "N/A"      → fallback
    """
    if value is None:
        return fallback
    try:
        result = float(str(value).strip())
        return result if result == result else fallback  # NaN guard
    except (ValueError, TypeError):
        return fallback


def _parse_candle(raw: dict, symbol: str = "") -> dict:
    """
    Parse a single Twelve Data candle from the 'values' array.

    Raw format (all strings):
    {
        "datetime": "2021-08-26 15:59:00",
        "open":     "148.372",
        "high":     "148.405",
        "low":      "148.365",
        "close":    "148.390",
        "volume":   "2206"        ← string, may be "0" or absent for forex
    }

    Returns a properly typed dict with a guaranteed 'volume' float key.
    """
    volume_raw = raw.get("volume")

    # Volume handling:
    # - Forex pairs: "0" is common (tick volume not always provided)
    # - Missing key: treat as 0 and let the engine handle gracefully
    volume = _safe_float(volume_raw, fallback=VOLUME_FALLBACK)

    return {
        "datetime": raw.get("datetime", ""),
        "open":     _safe_float(raw.get("open")),
        "high":     _safe_float(raw.get("high")),
        "low":      _safe_float(raw.get("low")),
        "close":    _safe_float(raw.get("close")),
        "volume":   volume,
    }


def _parse_candles(values: list, symbol: str = "") -> list:
    """
    Parse and order a full candle list from Twelve Data.

    Twelve Data returns values in DESCENDING order (newest first).
    We reverse to ascending (oldest first) for correct EMA/analysis.
    """
    parsed = [_parse_candle(c, symbol) for c in values]
    parsed.reverse()   # ascending: oldest → newest
    return parsed


def _assess_volume_quality(candles: list) -> dict:
    """
    Analyse the quality of volume data in a candle set.

    Returns a dict with:
      - has_volume:    bool — enough non-zero volume for spike analysis
      - zero_pct:      float — percentage of candles with zero volume
      - avg_volume:    float — mean of non-zero volumes
      - recommendation: str — human-readable quality note
    """
    volumes = [c["volume"] for c in candles]
    non_zero = [v for v in volumes if v > 0]
    zero_pct = (len(volumes) - len(non_zero)) / max(len(volumes), 1) * 100

    has_volume = len(non_zero) >= MIN_VOLUME_CANDLES
    avg_volume = sum(non_zero) / len(non_zero) if non_zero else 0.0

    if zero_pct >= 80:
        rec = "Volume unavailable for this asset — spike analysis disabled"
    elif zero_pct >= 40:
        rec = "Sparse volume data — spike signals may be unreliable"
    else:
        rec = "Volume data adequate for spike analysis"

    return {
        "has_volume":      has_volume,
        "zero_pct":        zero_pct,
        "avg_volume":      avg_volume,
        "recommendation":  rec,
    }

# TWELVE DATA FEED
# ─────────────────────────────────────────────────────────────────────────────

class TwelveDataFeed:
    """
    Async HTTP client for Twelve Data API.

    Provides:
      fetch_candles()      — closed M1 candles with volume, ascending order
      fetch_live_quote()   — forming (live) candle from /quote endpoint
      get_current_seconds_into_minute() — timing for pre-close detection
      validate_api_key()   — startup health check
    """

    BASE_URL = "https://api.twelvedata.com"

    def __init__(self, api_key: str):
        self.api_key = api_key
        self._session: Optional[aiohttp.ClientSession] = None

    async def _get_session(self) -> aiohttp.ClientSession:
        if self._session is None or self._session.closed:
            timeout = aiohttp.ClientTimeout(total=10)
            self._session = aiohttp.ClientSession(timeout=timeout)
        return self._session

    async def close(self):
        if self._session and not self._session.closed:
            await self._session.close()

    # ── CANDLE FETCH ──────────────────────────────────────────────────────────

    async def fetch_candles(
        self,
        symbol:   str,
        interval: str = "1min",
        count:    int = 60,
    ) -> list:
        """
        Fetch `count` closed M1 candles for `symbol`.

        Returns a list of candle dicts ordered oldest→newest:
        [
            {"datetime": ..., "open": float, "high": float,
             "low": float, "close": float, "volume": float},
            ...
        ]

        Volume notes:
        - Stocks/Crypto/Commodities: real traded volume (float)
        - Forex: tick volume (number of price ticks per minute)
        - Both are valid for relative spike detection in CRT analysis
        - Volume "0" values are preserved (engine handles gracefully)
        """
        session = await self._get_session()

        params = {
            "symbol":     symbol,
            "interval":   interval,
            "outputsize": count,
            "apikey":     self.api_key,
            "order":      "desc",   # newest first (we reverse after parsing)
            "format":     "JSON",
        }

        try:
            async with session.get(f"{self.BASE_URL}/time_series", params=params) as resp:
                if resp.status != 200:
                    logger.error(f"[{symbol}] HTTP {resp.status} from Twelve Data")
                    return []

                data = await resp.json()

                # API-level error check
                if data.get("status") == "error":
                    msg = data.get("message", "Unknown error")
                    logger.error(f"[{symbol}] Twelve Data error: {msg}")
                    return []

                values = data.get("values", [])
                if not values:
                    logger.warning(f"[{symbol}] No candle data returned")
                    return []

                # Parse with type coercion + ascending sort
                candles = _parse_candles(values, symbol)

                # Log volume quality for diagnostics
                vq = _assess_volume_quality(candles)
                if vq["zero_pct"] > 40:
                    logger.warning(
                        f"[{symbol}] Volume quality: {vq['zero_pct']:.0f}% zero — "
                        f"{vq['recommendation']}"
                    )
                else:
                    logger.debug(
                        f"[{symbol}] Volume quality OK — avg={vq['avg_volume']:.0f}, "
                        f"zero={vq['zero_pct']:.0f}%"
                    )

                return candles

        except asyncio.TimeoutError:
            logger.error(f"[{symbol}] Timeout fetching candles")
            return []
        except Exception as e:
            logger.error(f"[{symbol}] fetch_candles error: {e}")
            return []

    # ── LIVE QUOTE (FORMING CANDLE) ───────────────────────────────────────────

    async def fetch_live_quote(self, symbol: str) -> Optional[dict]:
        """
        Fetch the current live/forming candle from the /quote endpoint.

        Twelve Data /quote returns the CURRENT open candle, giving us
        a real-time snapshot for pre-close analysis.

        Returns a candle dict with volume (may be partial — that's expected).
        """
        session = await self._get_session()

        params = {
            "symbol": symbol,
            "apikey": self.api_key,
        }

        try:
            async with session.get(f"{self.BASE_URL}/quote", params=params) as resp:
                if resp.status != 200:
                    logger.error(f"[{symbol}] HTTP {resp.status} from /quote")
                    return None

                data = await resp.json()

                if data.get("status") == "error":
                    logger.error(f"[{symbol}] Quote error: {data.get('message')}")
                    return None

                # /quote response uses same field names — all strings
                # Fields: open, high, low, close, volume, previous_close, etc.
                return {
                    "datetime": data.get("datetime", ""),
                    "open":     _safe_float(data.get("open")),
                    "high":     _safe_float(data.get("high")),
                    "low":      _safe_float(data.get("low")),
                    "close":    _safe_float(data.get("close")),
                    "volume":   _safe_float(data.get("volume"), fallback=VOLUME_FALLBACK),
                }

        except asyncio.TimeoutError:
            logger.error(f"[{symbol}] Timeout on /quote")
            return None
        except Exception as e:
            logger.error(f"[{symbol}] fetch_live_quote error: {e}")
            return None

    # ── TIMING ────────────────────────────────────────────────────────────────

    async def get_current_seconds_into_minute(self) -> int:
        """
        Returns how many seconds into the current minute we are (0–59).
        Used by the CRT engine's pre-close detection (40–55s window).

        Based on UTC time to match Twelve Data's candle timestamps.
        """
        return datetime.now(timezone.utc).second

    # ── API KEY VALIDATION ────────────────────────────────────────────────────

    async def validate_api_key(self) -> str:
        """
        Validate the API key by hitting the /api_usage endpoint.

        Returns one of three strings:
          "ok"           — key is valid and has credits available
          "rate_limited" — key is valid but hit per-minute credit cap
                           (common on free tier if bot ran recently)
          "invalid"      — key is wrong, expired, or not recognised
        """
        session = await self._get_session()
        try:
            async with session.get(
                f"{self.BASE_URL}/api_usage",
                params={"apikey": self.api_key}
            ) as resp:
                data = await resp.json()

                # HTTP 200 + status ok = all good
                if resp.status == 200 and data.get("status") == "ok":
                    return "ok"

                # Rate limit message — key is valid, just throttled
                message = data.get("message", "") or data.get("code", "")
                if "run out of API credits" in str(message) or "429" in str(resp.status):
                    logger.warning(f"API key validation: rate limited — {message}")
                    return "rate_limited"

                # Anything else (401, wrong key, etc.) = invalid
                logger.error(f"API key validation failed: {message}")
                return "invalid"

        except Exception as e:
            logger.error(f"API key validation error: {e}")
            # Network error — assume rate limited not invalid (safer default)
            return "rate_limited"


# ─────────────────────────────────────────────────────────────────────────────
# VOLUME QUALITY CHECKER — call this at startup to verify your data
# ─────────────────────────────────────────────────────────────────────────────

async def check_volume_availability(
    feed:    TwelveDataFeed,
    symbols: list[str],
) -> dict[str, dict]:
    """
    Run a volume quality check across a list of symbols at startup.

    Usage in bot.py:
        report = await check_volume_availability(self.feed, ["EUR/USD", "BTC/USD", ...])
        for symbol, info in report.items():
            logger.info(f"{symbol}: {info['recommendation']}")

    Returns a dict of symbol → volume quality info.
    """
    results = {}
    for symbol in symbols:
        candles = await feed.fetch_candles(symbol, count=20)
        if candles:
            results[symbol] = _assess_volume_quality(candles)
        else:
            results[symbol] = {
                "has_volume":      False,
                "zero_pct":        100.0,
                "avg_volume":      0.0,
                "recommendation":  "Could not fetch candles",
            }
    return results
